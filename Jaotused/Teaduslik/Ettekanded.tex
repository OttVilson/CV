\newcommand\maaraAasta[1]{%
	\edef\praeguneAasta{#1}%
	\ifx\kontrollAasta\praeguneAasta%	
	\makebox[\widthof{\praeguneAasta}]{}
	\else%
	\praeguneAasta%
	\xdef\kontrollAasta{#1}%
	\fi%
}

\newcommand\kuiETISeKirjeOlemas[1]{%
	\edef\kontroll{\csname ettekanne\rom{#1}ETIS\endcsname}
	\ifx\kontroll\empty%
	\else%
	{\hypersetup{urlcolor = ETIScolor}%
		[\href{https://www.etis.ee/Portal/Publications/Display/\laienda{ettekanne\rom{#1}ETIS}}{ETIS}]%
	}%
	\fi%
}

	\phantomsection%
	\label{ettekanne\rom{#1}}%
	
	\addcontentsline{toc}{subsubsection}{\texorpdfstring{\hyperref[ettekanne\rom{#1}]{\laienda{ettekanne\rom{#1}jarjekord}) \laienda{ettekanne\rom{#1}pealkiri}}}{\laienda{ettekanne\rom{#1}jarjekord}) \laienda{ettekanne\rom{#1}PDFpealkiri}}}
	
	\begin{minipage}[t]{1.3cm}%
		\maaraAasta{\csname ettekanne\rom{#1}aasta\endcsname}%
	\end{minipage}%
	%
	\begin{minipage}[t]{0.7cm}%
		\laienda{ettekanne\rom{#1}jarjekord})%
	\end{minipage}%
	%
	\begin{minipage}[t]{0.845\textwidth}%
		\laienda{ettekanne\rom{#1}pealkiri}~-- %
		\laienda{ettekanne\rom{#1}konverentsinimi}, %
		\laienda{ettekanne\rom{#1}toimumiskoht} %
		\kuiETISeKirjeOlemas{#1}%
	\end{minipage}%
}
